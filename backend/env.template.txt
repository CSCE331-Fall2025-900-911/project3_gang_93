# ================ Environment Configuration Template ================
# Copy this file to .env and replace the values with your actual database credentials

# ================ Database Configuration ================
DB_NAME=your_database_name                # Database Name
DB_USER=your_username                     # Database User
DB_PASSWORD=your_password_here            # Database Password (REQUIRED - add your actual password here)
DB_HOST=your_database_host                # Database Host
DB_PORT=5432                              # Database Port (default: 5432 for PostgreSQL)

# ================ CORS Configuration For Accepted Frontend URL's ================
CORS_ORIGINS="your_frontend_url1.com,your_frontend_url2.net"      # The base url for your website (no "/" - extensions not needed)

# ================ Google OAuth Configuration ================
GOOGLE_CLIENT_ID=your_google_client_id        # The ID google gives you
GOOGLE_CLIENT_SECRET=your_clients_secret      # The Secret that is given upon creation of the client
GOOGLE_REDIRECT_URI=you_webAppRedirect        # Where you want authorized users to be redirected
